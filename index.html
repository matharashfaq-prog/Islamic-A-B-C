// Feature 2: Core Calculations with Dynamic Weight, Baking Loss & Crumb Wastage function manualCalculate() { let budget = parseFloat(document.getElementById('budgetInput').value) || 0; let singleKhataiWeight = parseFloat(document.getElementById('khataiWeightInput').value) || 15; let rGhee = parseFloat(document.getElementById('rateGhee').value) || 0; let rSugar = parseFloat(document.getElementById('rateSugar').value) || 0; let rFlour = parseFloat(document.getElementById('rateFlour').value) || 0; let rSuji = parseFloat(document.getElementById('rateSuji').value) || 0; let rBP = parseFloat(document.getElementById('rateBP').value) || 0; // 1 KG Raw formulation cost let rawMaterialCostPerKG = (rGhee * 0.3) + (rSugar * 0.3) + (rFlour * 0.3) + (rSuji * 0.1) + (rBP * (BP_PER_BATCH_G/1000)); let totalFactoryCostPerKG = rawMaterialCostPerKG + LABOUR_FIXED + GAS_FIXED; // Baking effect (8% moisture loss + 5% breakages = 13% total drop) let finalYieldMultiplier = 0.87; let totalRawKGCanBuy = budget / totalFactoryCostPerKG; let readyKhataiGrams = (totalRawKGCanBuy * 1000) * finalYieldMultiplier; let totalGoodKhatais = readyKhataiGrams / singleKhataiWeight; let totalBoxes = totalGoodKhatais / 24; let singleBoxWeight = singleKhataiWeight * 24; // Tier based Carton discount check let wholesalepriceBox = (budget >= 10000) ? 190 : 200; // Dynamic Costs per entities let singleCost = totalFactoryCostPerKG / (1000 * finalYieldMultiplier / singleKhataiWeight); let boxCost = (singleCost * 24) + BOX_PACKING_COST; let boxCostWithCommission = boxCost + RIDER_COMMISSION; let cartonCost = boxCostWithCommission * 24; let cartonProfit = (wholesalepriceBox - boxCostWithCommission) * 24; // Update UI Outputs document.getElementById('resSingleCost').innerText = singleCost.toFixed(2) + " روپے"; document.getElementById('resBoxCost').innerText = boxCostWithCommission.toFixed(2) + " روپے"; document.getElementById('resCartonCost').innerText = cartonCost.toFixed(2) + " روپے"; document.getElementById('resCartonProfit').innerText = cartonProfit.toFixed(2) + " روپے"; document.getElementById('tableRawWeight').innerText = totalRawKGCanBuy.toFixed(2) + " کلو گرام"; document.getElementById('tableTotalBoxes').innerText = Math.floor(totalBoxes) + " ڈبے"; document.getElementById('tableBoxNetWeight').innerText = singleBoxWeight + " گرام"; profitChart.data.datasets[0].data = [totalProfit * 0.2, totalProfit * 0.5, totalProfit, totalProfit * 4.5]; profitChart.update(); } } function initChart() { let ctx = document.getElementById('profitChart').getContext('2d'); profitChart = new Chart(ctx, { type: 'bar', data: { labels: ['روزانہ', 'ہفتہ وار', 'ماہانہ', 'سالانہ (تخمینہ)'], datasets: [{ label: 'خالص منافع کا گراف (روپے)', data: [0, 0, 0, 0], backgroundColor: '#d97706', borderRadius: 8 }] }, options: { responsive: true, scales: { y: { beginAtZero: true } } } }); updateReports(); }
